.text
.org 0x20
@spremenljivke
stevilo_ponovitev_notranje: .word 48000
.align
.global _start
_start:
@program
	adr r0, stevilo_ponovitev_notranje
	mov r1, #0
	mov r2, #500
ZUNANJA_ZANKA:
	ldr r3, [r0]
NOTRANJA_ZANKA:
		add r1, r1, #1
		subs r3, r3, #1
		bne NOTRANJA_ZANKA
	subs r2, r2, #1
	bne ZUNANJA_ZANKA
end: b end

4. naloga

.text
.org 0x20
@spremenljivke
niz: .asciz "Zunaj je mraz, sneg in poledica."
.align
.global _start
_start:
@program
	adr r0, niz
	mov r1, #0
	mov r2, #0x20
zanka:
	ldrb r3, [r0]
	cmp r3, #0
	beq konec
	cmp r3, r2
	addeq r1, r1, #1
	add r0, r0, #1
	b zanka
konec:
end: b end
